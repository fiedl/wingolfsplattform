def editable_span_tag( params )

  urls = {}
  [ 'show', 'edit', 'update', 'create', 'new' ].each do |action| 
    urls[ "#{action}_url" ] = url_for( params.merge( action: action ) )
  end
  
  content_tag( :span, "", { class: "editable" }.merge( urls ) )
  
end
