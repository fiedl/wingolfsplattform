def create
  @title = t :login
  begin
    @current_user = User.authenticate( params[ :login_name ], params[ :password ] )
    if @current_user
      @session.login @current_user
      redirect_to :controller => "users", :action => "show", :alias => @current_user.alias
    end
  rescue => exception
    flash[ :error ] = t exception
    render :action => "new"
  end
end
